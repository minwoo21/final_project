Javascript Callback 예제

2019-07-29 13:06
By JaewoongByeon
Preview
Code 

<!DOCTYPE html>
<html>

<head>
    <script>
        //function 3개로 callback
        /*
        function execFunction(param) {
            prefunction(param.name, callbackfunction);
        };
        function prefunction(param02, callback01) {
            if (typeof callback01 == 'function') {
                callback01(param02);
            }
        }
        function callbackfunction(param01) {
            document.getElementById("result").innerHTML = "You clicked " + param01 + "!"
        }*/
        //3 functions in 1 object
        var exec = {
            execFunction: function (param) {
                this.prefunction(param.name, this.callbackfunction);
            },
            prefunction: function (param02, callback01) {
                if (typeof callback01 == 'function') {
                    callback01(param02);
                }
            },
            callbackfunction: function (param01) {
                document.getElementById("result").innerHTML = "You clicked " + param01 + "!"
            }
        };
    </script>
</head>

<body>
    <button onclick="exec.execFunction(this);" name="Button01">Button01</button>
    <button onclick="exec.execFunction(this);" name="Button02">Button02</button>
    <p id="result"></p>
</body>

</html>
